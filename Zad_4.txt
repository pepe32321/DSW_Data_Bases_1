SELECT NAZWISKO,ZATRUDNIONY, 
CASE 
    WHEN EXTRACT(YEAR FROM ZATRUDNIONY) IN (1992 , 1996) THEN CAST(ROUND(MONTHS_BETWEEN(CURRENT_DATE,ZATRUDNIONY),0) AS VARCHAR(20))
	ELSE CASE
    	WHEN MONTHS_BETWEEN('01-JAN-2023',ZATRUDNIONY) > 400 THEN 'Długi'
    	ELSE 'Krótki'
    END
END AS "Staż pracy"
FROM PRACOWNICY

ORDER BY MONTHS_BETWEEN(CURRENT_DATE,ZATRUDNIONY)
