SELECT NAZWISKO,ZATRUDNIONY,CAST(ROUND(MONTHS_BETWEEN(CURRENT_DATE,ZATRUDNIONY),0) AS VARCHAR(20)) AS "Staż pracy"
FROM PRACOWNICY
WHERE EXTRACT(YEAR FROM ZATRUDNIONY) IN (1992 , 1996)
UNION
SELECT NAZWISKO,ZATRUDNIONY,'Długi' AS "Staż pracy"
FROM PRACOWNICY
WHERE MONTHS_BETWEEN('01-JAN-2023',ZATRUDNIONY) > 400 AND EXTRACT(YEAR FROM ZATRUDNIONY) NOT IN (1992 , 1996)
UNION
SELECT NAZWISKO,ZATRUDNIONY,'Krótki' AS "Staż pracy"
FROM PRACOWNICY
WHERE MONTHS_BETWEEN('01-JAN-2023',ZATRUDNIONY) < 400 AND EXTRACT(YEAR FROM ZATRUDNIONY) NOT IN (1992 , 1996)

ORDER BY ZATRUDNIONY DESC
